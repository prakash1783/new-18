provider "aws" {
  region = "ap-south-1"
}

resource "aws_instance" "web" {
    count = var.instance_count
    ami = var.ami
    instance_type = var.instance_type
    tags = {
        name = var.tags
    }
}

variable "ami" {
    type = string
    default = "0f918f7e67a3323f0"
}

variable "instance_type" {
    type = string
    default = "t2.micro"
}

variable "tags" {
    type = string
    default = ""
}
variable "instance_count" {
    type = number
    default ="2"
}
